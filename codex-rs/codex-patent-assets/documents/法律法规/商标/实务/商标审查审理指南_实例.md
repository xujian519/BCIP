**使用方式**：
```python
import json

with open('instances_index.json') as f:
    data = json.load(f)

# 查找有OCR文字的实例
with_ocr = [inst for inst in data['instances'] 
            if inst.get('ocr_text') and inst['ocr_text'].strip()]

# 按页码分组
by_page = {}
for inst in with_ocr:
    page = inst['page']
    if page not in by_page:
        by_page[page] = []
    by_page[page].append(inst)
```
