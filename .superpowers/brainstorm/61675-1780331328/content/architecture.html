状态设计

struct SettingsPanel {
    active_tab: SettingsTab,     // General | Model | Permissions | Appearance
    items: Vec,    // 当前页签的设置项列表
    focus_index: usize,          // 当前焦点位置
}

enum SettingsTab {
    General,
    Model,
    Permissions,
    Appearance,
}

struct SettingsItem {
    label: String,
    description: String,
    kind: SettingsItemKind,      // Toggle | Select | Action
    enabled: bool,
}
